00022         inline static void QuaternionToEuler(const btQuaternion &TQuat, btVector3 &TEuler)
00023         {
00024            btScalar W = TQuat.getW();
00025            btScalar X = TQuat.getX();
00026            btScalar Y = TQuat.getY();
00027            btScalar Z = TQuat.getZ();
00028            float WSquared = W * W;
00029            float XSquared = X * X;
00030            float YSquared = Y * Y;
00031            float ZSquared = Z * Z;
00032            TEuler.setX(atan2f(2.0f * (Y * Z + X * W), -XSquared - YSquared + ZSquared + WSquared));
00033            TEuler.setY(asinf(-2.0f * (X * Z - Y * W)));
00034            TEuler.setZ(atan2f(2.0f * (X * Y + Z * W), XSquared - YSquared - ZSquared + WSquared));
00035            TEuler *= RADTODEG;
00036         }
